							// Tesla.java
							class Tesla {
								String owner = "Tyler";
								String color = "red";
								int Doors = 4;
								public String getOwner() {
									return owner;
								}
							
								public void setOwner(String owner) {
									this.owner = owner;
								}
							
								public String getColor() {
									return color;
								}
							
								public void setColor(String color) {
									this.color = color;
								}
							
								public int getDoors() {
									return Doors;
								}
							
								public void setDoors(int doors) {
									Doors = doors;
								}
							}
